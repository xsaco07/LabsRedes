=== RStick 
RStick>ena
RStick#conf t
RStick(config)#interface fastEthernet 0/0
RStick(config-if)#no sh
RStick(config-if)#interface fastEthernet 0/0.10
RStick(config-subif)#encapsulation dot1Q 10
RStick(config-subif)#ip address 172.168.10.1 255.255.255.0

RStick(config-subif)#interface fastEthernet 0/0.20
RStick(config-subif)#encapsulation dot1Q 20
RStick(config-subif)#ip address 172.168.20.1 255.255.255.0
RStick(config-subif)#exit

RStick(config)#ip dhcp excluded-address 172.168.10.1
RStick(config)#ip dhcp excluded-address 172.168.20.1

RStick(config)#ip dhcp pool vlan10
RStick(dhcp-config)#network 172.168.10.0 255.255.255.0
RStick(dhcp-config)#default-router 172.168.10.1
RStick(dhcp-config)#dns-server 196.168.10.2

RStick(dhcp-config)#ip dhcp pool vlan20
RStick(dhcp-config)#dns-server 196.168.10.2
RStick(dhcp-config)#default-router 172.168.20.1
RStick(dhcp-config)#network 172.168.20.0 255.255.255.0
RStick(dhcp-config)#exit

RStick(config)#router ospf 10
RStick(config-router)#network 172.168.10.0 0.0.0.255 area 0
RStick(config-router)#network 172.168.20.0 0.0.0.255 area 0
RStick(config-router)#network 172.168.0.0 0.0.0.3 area 0
